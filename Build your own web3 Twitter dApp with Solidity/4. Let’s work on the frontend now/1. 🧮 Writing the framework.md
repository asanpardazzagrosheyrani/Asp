export default class EthereumRpc {

  private provider: SafeEventEmitterProvider;

    this.provider = provider;

  }

  async getAccounts(): Promise<string[]> {

    try {

      const web3 = new Web3(this.provider as any);

      const accounts = await web3.eth.getAccounts();

      return accounts;

    } catch (error: unknown) {

      return error as string[];

    }

  }

    try {

      const alchemyKey = APP_CONSTANTS.ALCHEMY_KEY;

Wait a minute! Sit and relax. Don‚Äôt get overwhelmed by the entire code upfront. Let‚Äôs look at individual functions one at a time and see what it is responsible for.

The first main function that we have is getAccounts(). This function is responsible for getting the account address of the logged in user. Why do we need this account address? That‚Äôs a good question. This account address is required whenever we want to change something to the state of contract. For example, if you want to add a new tweet, you will have to specify the account address. Otherwise, your request will either be rejected or failed with an error.
